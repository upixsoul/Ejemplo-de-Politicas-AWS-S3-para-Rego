Ejemplo de Políticas AWS S3 para Gestión de TI (AL07106842 David Garcia)

Para probar utiliza:

https://play.openpolicyagent.org

Este manual utiliza el enfoque de Whitelist/Permitir. La respuesta de OPA siempre será un booleano claro (true/false), facilitando la integración con servicios externos.

1. Cifrado Obligatorio (AES256)

package aws.s3
default allow := false

allow if {
    some conf in input.resource.aws_s3_bucket.server_side_encryption_configuration
    some rule in conf.rule
    some enc in rule.apply_server_side_encryption_by_default
    enc.sse_algorithm == "AES256"
}
Escenario: Válido (Cifrado Activo) 
{"resource":{"aws_s3_bucket":{"server_side_encryption_configuration":[{"rule":[{"apply_server_side_encryption_by_default":[{"sse_algorithm":"AES256"}]}]}]}}}
Resultado OPA: { "allow": true }
Escenario: Inválido (Sin Cifrado) 
{"resource":{"aws_s3_bucket":{"server_side_encryption_configuration":[]}}}
Resultado OPA: { "allow": false }

2. Bloqueo de ACL Pública

package aws.s3
default allow := false

# Permitir solo si el ACL es estrictamente 'private'
allow if {
    some _, bucket in input.resource.aws_s3_bucket
    bucket.acl == "private"
}
Escenario: Válido (Privado) 
{"resource":{"aws_s3_bucket":{"b1":{"acl":"private"}}}}
Resultado OPA: { "allow": true }
Escenario: Inválido (Público) 
{"resource":{"aws_s3_bucket":{"b1":{"acl":"public-read"}}}}
Resultado OPA: { "allow": false }

3. Región Permitida (us-east-1)

package aws.s3
default allow := false

allow if {
    some _, bucket in input.resource.aws_s3_bucket
    bucket.region == "us-east-1"
}
Escenario: Región Correcta 
{"resource":{"aws_s3_bucket":{"b1":{"region":"us-east-1"}}}}
Resultado OPA: { "allow": true }
Escenario: Región Prohibida 
{"resource":{"aws_s3_bucket":{"b1":{"region":"us-west-2"}}}}
Resultado OPA: { "allow": false }

4. Requerir HTTPS (SecureTransport)

package aws.s3
default allow := false

allow if {
    some _, p in input.resource.aws_s3_bucket_policy
    contains(p.policy, "aws:SecureTransport")
}
Escenario: Política con HTTPS 
{"resource":{"aws_s3_bucket_policy":{"p1":{"policy":"{\"Condition\":{\"Bool\":{\"aws:SecureTransport\":\"true\"}}}"}}}}
Resultado OPA: { "allow": true }
Escenario: Política insegura 
{"resource":{"aws_s3_bucket_policy":{"p1":{"policy":"{}"}}}}
Resultado OPA: { "allow": false }

5. Etiquetado FinOps (Project)

package aws.s3
default allow := false

allow if {
    some _, bucket in input.resource.aws_s3_bucket
    bucket.tags.Project
}
Escenario: Con Etiqueta 
{"resource":{"aws_s3_bucket":{"b1":{"tags":{"Project":"Editorial"}}}}}
Resultado OPA: { "allow": true }
Escenario: Sin Etiqueta 
{"resource":{"aws_s3_bucket":{"b1":{"tags":{"Env":"Prod"}}}}}
Resultado OPA: { "allow": false }

6. Versionado Obligatorio

package aws.s3
default allow := false

allow if {
    some _, bucket in input.resource.aws_s3_bucket
    some v in bucket.versioning
    v.enabled == true
}
Escenario: Versionado Activo 
{"resource":{"aws_s3_bucket":{"b1":{"versioning":[{"enabled":true}]}}}}
Resultado OPA: { "allow": true }
Escenario: Versionado Apagado 
{"resource":{"aws_s3_bucket":{"b1":{"versioning":[{"enabled":false}]}}}}
Resultado OPA: { "allow": false }

7. Nomenclatura Estándar (Sin 'test')

package aws.s3
default allow := true  # Por defecto permitimos

# Pero revocamos si el nombre contiene 'test'
allow := false if {
    some name, _ in input.resource.aws_s3_bucket
    contains(name, "test")
}
Escenario: Nombre Producción 
{"resource":{"aws_s3_bucket":{"api-produccion":{}}}}
Resultado OPA: { "allow": true }
Escenario: Nombre de Prueba 
{"resource":{"aws_s3_bucket":{"test-endpoint":{}}}}
Resultado OPA: { "allow": false }
